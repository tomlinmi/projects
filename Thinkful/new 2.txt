function expandArticleBody() { 
  const allBtns = document.querySelectorAll(".expand_button");
  for(let btn of allBtns){ btn.addEventListener("click", (event) => { 
    let x = event.target.parentNode.parentNode.parentNode.querySelector(".article_body"); 
    if(btn.innerText == "V"){ x.parentNode.style.display = "none"; btn.innerText = ">"; } 
    else { 
      x.style.display = "block"; btn.innerText = 'V'; } 
  }) 
 } 
}
